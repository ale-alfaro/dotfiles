lastOrNull

kotlin-stdlib / kotlin.ranges / lastOrNull
Platform and version requirements: JVM (1.7), JS (1.7), Native (1.7) 
fun IntProgression.lastOrNull(): Int?
fun LongProgression.lastOrNull(): Long?
fun CharProgression.lastOrNull(): Char?
fun UIntProgression.lastOrNull(): UInt?
fun ULongProgression.lastOrNull(): ULong?

Returns the last element, or null if the progression is empty.

import kotlin.test.*

fun main(args: Array<String>) {
//sampleStart
val list = listOf(1, 2, 3, 4)
println(list.last()) // 4
println(list.last { it % 2 == 1 }) // 3
println(list.lastOrNull { it < 0 }) // null
// list.last { it < 0 } //  will fail

val emptyList = emptyList<Int>()
println(emptyList.lastOrNull()) // null
// emptyList.last() //  will fail
//sampleEnd
}
